cluster_purity = function(cluster, truth){
  comp = as_tibble(cbind(cluster,truth))
  purity = comp %>%
    group_by(truth) %>% 
    count(cluster) %>% 
    summarise(purity = max(n)/sum(n)) %>% 
    pull(purity)
  return(purity)
}
